🎮 2. Shared Game Engine (VERY IMPORTANT)
📄 hooks/useGame.js
import { useReducer } from "react";

const initial = {
  word: "",
  guessed: [],
  wrong: 0,
  score: 0,
  streak: 0
};

function reducer(state, action) {
  switch (action.type) {
    case "SET_WORD":
      return { ...state, word: action.word };

    case "CORRECT":
      return {
        ...state,
        guessed: [...state.guessed, action.letter],
        streak: state.streak + 1,
        score: state.score + 10 * (state.streak + 1)
      };

    case "WRONG":
      return {
        ...state,
        guessed: [...state.guessed, action.letter],
        wrong: state.wrong + 1,
        streak: 0
      };

    default:
      return state;
  }
}

export default function useGame() {
  return useReducer(reducer, initial);
}

📅 3. Daily Challenge
📄 components/DailyChallenge.js
function getDailyWord(words) {
  const today = new Date().toDateString();
  const hash = [...today].reduce((a, c) => a + c.charCodeAt(0), 0);
  return words[hash % words.length];
}

👥 4. Multiplayer Rooms
📄 services/realtime.js
import { ref, set, onValue } from "firebase/database";
import { rtdb } from "./firebase";

export function joinRoom(roomId, user) {
  set(ref(rtdb, `rooms/${roomId}/players/${user.uid}`), {
    name: user.displayName,
    score: 0
  });
}

export function listenRoom(roomId, callback) {
  onValue(ref(rtdb, `rooms/${roomId}`), snap => {
    callback(snap.val());
  });
}

export async function share(id) {
